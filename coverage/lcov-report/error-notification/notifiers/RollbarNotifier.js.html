  /**
   * Send error to Rollbar.
   * @param {Object} error - Error value.
   * @param {String} correlationId - Rollbar correlation id.
   */
  sendErrorNotification(error, correlationId) {
    instance.error(`Correlation ID: ${correlationId} ${error.toString()}`);
    instance.wait(() => {
      throw error;
    });
  }
